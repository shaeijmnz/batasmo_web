-- 8. RPC FUNCTION FOR BOOKING APPOINTMENTS
CREATE OR REPLACE FUNCTION book_appointment(
  client_uuid UUID, 
  attorney_uuid UUID, 
  scheduled_time TIMESTAMP, 
  title_text TEXT
) RETURNS void AS $$
BEGIN
  -- Insert appointment
  INSERT INTO appointments (client_id, attorney_id, scheduled_at, title, status)
  VALUES (client_uuid, attorney_uuid, scheduled_time, title_text, 'confirmed');
  
  -- Update availability slot
  UPDATE availability_slots 
  SET is_booked = true 
  WHERE attorney_id = attorney_uuid 
  AND date = scheduled_time::date 
  AND time = to_char(scheduled_time, 'HH:MI AM');
END;
$$ LANGUAGE plpgsql;

-- 2. RPC to log a failed attempt (Bypasses RLS to allow the system to write to it)
CREATE OR REPLACE FUNCTION log_failed_login(user_email TEXT)
RETURNS void AS $$
BEGIN
    INSERT INTO audit_logs (email) VALUES (user_email);
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 3. RPC to check if locked out (3 strikes in 15 mins)
CREATE OR REPLACE FUNCTION check_login_lockout(user_email TEXT)
RETURNS boolean AS $$
DECLARE
    recent_fails INT;
BEGIN
    SELECT COUNT(*) INTO recent_fails
    FROM audit_logs
    WHERE email = user_email
    AND attempt_time >= NOW() - INTERVAL '15 minutes';
    
    RETURN recent_fails >= 3;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- 4. RPC to clear strikes upon successful login
CREATE OR REPLACE FUNCTION clear_failed_logins(user_email TEXT)
RETURNS void AS $$
BEGIN
    DELETE FROM audit_logs WHERE email = user_email;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

ALTER TABLE profiles 
  ADD CONSTRAINT chk_full_name_len CHECK (char_length(full_name) <= 100),
  ADD CONSTRAINT chk_email_len CHECK (char_length(email) <= 150),

-- 1. Drop the old boolean version
DROP FUNCTION IF EXISTS check_login_lockout(text);
-- 2. Create the new dynamic integer version
CREATE OR REPLACE FUNCTION check_login_lockout(user_email TEXT)
RETURNS int AS $$
DECLARE
    recent_fails INT;
    last_attempt TIMESTAMP WITH TIME ZONE;
    seconds_remaining INT;
BEGIN
    SELECT COUNT(*), MAX(attempt_time)
    INTO recent_fails, last_attempt
    FROM audit_logs
    WHERE email = user_email
    AND attempt_time >= NOW() - INTERVAL '15 minutes';
    
    IF recent_fails >= 3 THEN
        seconds_remaining := EXTRACT(EPOCH FROM (last_attempt + INTERVAL '15 minutes' - NOW()));
        IF seconds_remaining > 0 THEN
            RETURN seconds_remaining;
        END IF;
    END IF;
    
    RETURN 0;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- Auto-create profile when a new auth user is created
create or replace function public.handle_new_user () RETURNS trigger LANGUAGE plpgsql SECURITY DEFINER
set
  search_path = public as $$
BEGIN
  INSERT INTO public.profiles (id, email, full_name, role)
  VALUES (
    NEW.id,
    NEW.email,
    COALESCE(NEW.raw_user_meta_data ->> 'full_name', ''),
    COALESCE((NEW.raw_user_meta_data ->> 'role')::user_role, 'Client')
  )
  ON CONFLICT (id) DO UPDATE
  SET
    email = EXCLUDED.email,
    full_name = COALESCE(NULLIF(EXCLUDED.full_name, ''), public.profiles.full_name),
    role = EXCLUDED.role,
    updated_at = now();

  RETURN NEW;
END;
$$;

-- Run after 20260406_reset_seed_attorney_passwords.sql

WITH target_emails AS (
  SELECT unnest(ARRAY[
    'test.jeanne.castillo-anarna@batasmo.app',
    'test.alston.anarna@batasmo.app',
    'test.allen.anarna@batasmo.app'
  ]::text[]) AS email
),
auth_rows AS (
  SELECT
    lower(u.email) AS email,
    u.id AS auth_user_id,
    u.email_confirmed_at,
    u.created_at AS auth_created_at,
    u.updated_at AS auth_updated_at
  FROM auth.users u
),
identity_rows AS (
  SELECT
    lower(i.provider_id) AS email,
    i.user_id,
    i.provider,
    i.created_at AS identity_created_at,
    i.updated_at AS identity_updated_at
  FROM auth.identities i
  WHERE i.provider = 'email'
),
profile_rows AS (
  SELECT
    lower(p.email) AS email,
    p.id AS profile_id,
    p.role,
    p.full_name,
    p.updated_at AS profile_updated_at
  FROM public.profiles p
)
SELECT
  t.email,
  a.auth_user_id,
  p.profile_id,
  (a.auth_user_id IS NOT NULL) AS has_auth_user,
  (i.user_id IS NOT NULL) AS has_email_identity,
  (p.profile_id IS NOT NULL) AS has_profile,
  (a.auth_user_id IS NOT NULL AND p.profile_id IS NOT NULL AND a.auth_user_id = p.profile_id) AS auth_profile_id_match,
  (i.user_id IS NOT NULL AND a.auth_user_id IS NOT NULL AND i.user_id = a.auth_user_id) AS identity_user_match,
  p.role,
  p.full_name,
  a.email_confirmed_at,
  a.auth_updated_at,
  p.profile_updated_at
FROM target_emails t
LEFT JOIN auth_rows a ON a.email = t.email
LEFT JOIN identity_rows i ON i.email = t.email
LEFT JOIN profile_rows p ON p.email = t.email
ORDER BY t.email;
